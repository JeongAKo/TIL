    button.setImage(UIImage(named: "bookMarkOpa"), for: .normal)
    button.setImage(UIImage(named: "bookMarkOpaSelected"), for: .selected)

    addSubview(button)
    return button
  }()

    sender.isSelected.toggle()
  }
